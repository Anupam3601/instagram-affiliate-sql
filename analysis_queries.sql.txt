--all posts along with their media type and post date, ordered by most recent first.
SELECT
    post_id,
    caption,
    media_type,
    post_date
FROM posts
ORDER BY post_date DESC;

--List all distinct media types used across all posts
SELECT DISTINCT media_type
FROM posts
ORDER BY media_type;

--all affiliate products in the 'Beauty' category, ordered by price descending.
SELECT
    product_id,
    product_name,
    brand,
    price,
    commission_rate
FROM affiliate_products
WHERE category = 'Beauty'
ORDER BY price DESC;

--Find all link clicks that came from the 'story' source on mobile devices.
SELECT
    click_id,
    post_id,
    click_source,
    device_type,
    clicked_at
FROM link_clicks
WHERE click_source = 'story'
  AND device_type  = 'mobile'
ORDER BY clicked_at DESC;

--10 most recent confirmed conversions with their sale amount and commission.
SELECT
    conversion_id,
    post_id,
    sale_amount,
    commission,
    converted_at
FROM conversions
WHERE status = 'confirmed'
ORDER BY converted_at DESC
LIMIT 10;

--Count the total number of posts published for each media type.
SELECT
    media_type,
    COUNT(*) AS total_posts
FROM posts
GROUP BY media_type
ORDER BY total_posts DESC;

--What is the total number of views, total likes, and total comments recorded across all posts?
SELECT
    SUM(views)    AS total_views,
    SUM(likes)    AS total_likes,
    SUM(comments) AS total_comments,
    SUM(saves)    AS total_saves
FROM post_insights;

--List all affiliate products whose commission rate is greater than 10%, ordered by commission rate descending.

SELECT
    product_name,
    brand,
    category,
    commission_rate
FROM affiliate_products
WHERE commission_rate > 10
ORDER BY commission_rate DESC;

--Show each post along with its total views, likes, and comments from post_insights.
SELECT 
    p.post_date,
    p.post_id,
    p.post_url,
    p.media_type,
    pi.views,
    pi.comments,
    pi.likes,
    pi.saves,
    pi.reach
FROM posts AS p
JOIN post_insights AS pi
    ON p.post_id = pi.post_id
ORDER BY pi.views DESC;

--Calculate the engagement rate (likes + comments + saves) / views × 100 for every post. Show only posts with engagement rate above 3%.
SELECT *
FROM (
    SELECT 
        p.post_id,
        p.post_date,
        p.media_type,
        pi.views,
        ROUND(
            (pi.likes + pi.comments + pi.saves) * 100.0 
            / NULLIF(pi.views, 0), 
        2) AS engagement_rate
    FROM posts p
    JOIN post_insights pi
        ON p.post_id = pi.post_id
) 
WHERE engagement_rate > 3
ORDER BY engagement_rate DESC;

--Show each post with its affiliated product name and brand. Include posts even if no product is assigned.
SELECT 
    p.post_date,
    p.post_id,
    p.media_type,
    ap.product_name,
    ap.brand,
    ap.category
FROM posts AS p
JOIN affiliate_products AS ap
    ON p.affiliate_product_id = ap.product_id
ORDER BY p.post_date DESC;
--How many total clicks did each post receive? Show post_id, media_type, and total_clicks.
SELECT
    p.post_id,
    p.media_type,
    COUNT(lc.click_id) AS total_clicks
FROM posts p
LEFT JOIN link_clicks lc
    ON p.post_id = lc.post_id
GROUP BY p.post_id, p.media_type
ORDER BY total_clicks DESC;

--Find the total revenue and commission earned per affiliate product. Only include products with at least 1 confirmed conversion.
SELECT 
    ap.product_name,
    ap.brand,
    ap.category,
    SUM(c.sale_amount) AS Total_sale,
    SUM(c.commission) AS Total_Commosion
FROM affiliate_products ap
JOIN posts p
    ON ap.product_id = p.affiliate_product_id
JOIN conversions c
    ON c.post_id = p.post_id
WHERE c.status = 'confirmed'
GROUP BY 
    ap.product_id,
    ap.product_name,
    ap.brand,
    ap.category
HAVING COUNT(c.conversion_id) >= 1
ORDER BY Total_sale DESC;

--Show monthly totals of views, clicks, and conversions. Group by year and month.
SELECT
    DATE_FORMAT(p.post_date, '%Y-%m') AS month,
    SUM(pi.views)                       AS total_views,
    COUNT(DISTINCT lc.click_id)        AS total_clicks,
    COUNT(DISTINCT c.conversion_id)    AS total_conversions
FROM posts p
JOIN post_insights pi  ON p.post_id = pi.post_id
LEFT JOIN link_clicks lc ON p.post_id = lc.post_id
LEFT JOIN conversions c  ON lc.click_id = c.click_id
GROUP BY month
ORDER BY month ASC;
